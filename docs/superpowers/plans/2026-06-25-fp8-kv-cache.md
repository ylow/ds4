# FP8 (split) Compressed-Attention KV — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Store the 512-dim compressed-attention KV (`layer_attn_comp_cache`) as the model's E4M3 byte + per-64-group exponent for the 448 NoPE dims and F16 for the 64 RoPE dims, cutting that cache ~1.75× vs F16 (~3.5× vs F32), near-losslessly.

**Architecture:** Extend the proven **expand-on-read** path. The narrow cache is dequantized into the existing reused F32 scratch by new CUDA kernels; the trusted F32 attention kernels stay byte-identical. The write **commit** re-runs the model's own quantizer (`fp8_kv_quantize_kernel` math) but emits `(byte, exponent)` to narrow buffers + F16 RoPE — value-exact even on already-QAT'd staging. Gated by a new compile-time macro `DS4_GPU_ATTN_COMP_CACHE_FP8` that takes precedence over `_F16`.

**Tech Stack:** C + CUDA (nvcc), single file each: `ds4.c` (host/orchestration) and `ds4_cuda.cu` (kernels). No unit-test framework — the gate is the **teacher-forced perplexity oracle** (bit-deterministic).

```c
/* Decode a stored E4M3FN byte back to float (magnitude via the model table + sign).
 * bit7 = sign, bits0..6 = the magnitude index used by dsv4_e4m3fn_value_dev. */
__device__ static float dsv4_e4m3fn_decode_byte_dev(uint8_t b) {
    float mag = dsv4_e4m3fn_value_dev((int)(b & 0x7fu));
    return (b & 0x80u) ? -mag : mag;
}
```

- [ ] **Step 2: Add the 2 FP8 dequant kernels (ds4_cuda.cu, right after `gather_comp_f16_to_f32_kernel` at ~3722).**

```c
/* Expand all n_comp FP8-split rows to F32: NoPE = decode(byte) * 2^exp[group];
 * RoPE tail = f32(half).  group = d>>6 (d<n_nope), exp buffer is 8 int8/row. */
__global__ static void expand_comp_fp8_to_f32_kernel(
        float *out, const uint8_t *nope, const int8_t *expo, const __half *rope,
        uint32_t n_comp, uint32_t head_dim, uint32_t n_rot) {
    const uint64_t total = (uint64_t)n_comp * head_dim;
    uint64_t i = (uint64_t)blockIdx.x * blockDim.x + threadIdx.x;
    if (i >= total) return;
    const uint32_t row = (uint32_t)(i / head_dim);
    const uint32_t d   = (uint32_t)(i % head_dim);
    const uint32_t n_nope = head_dim - n_rot;
    if (d < n_nope) {
        const uint8_t b = nope[(uint64_t)row * n_nope + d];
        const int k = (int)expo[(uint64_t)row * 8u + (d >> 6)];
        out[i] = dsv4_e4m3fn_decode_byte_dev(b) * exp2f((float)k);
    } else {
        out[i] = __half2float(rope[(uint64_t)row * n_rot + (d - n_nope)]);
    }
}

/* Gather-of-selected: dequantize only the n_sel topk rows (decode path), out-of-range -> 0. */
__global__ static void gather_comp_fp8_to_f32_kernel(
        float *out, const uint8_t *nope, const int8_t *expo, const __half *rope,
        const int32_t *topk, uint32_t n_sel, uint32_t n_comp,
        uint32_t head_dim, uint32_t n_rot) {
    const uint64_t total = (uint64_t)n_sel * head_dim;
    uint64_t i = (uint64_t)blockIdx.x * blockDim.x + threadIdx.x;
    if (i >= total) return;
    const uint32_t orow = (uint32_t)(i / head_dim);
    const uint32_t d    = (uint32_t)(i % head_dim);
    const int32_t src = topk[orow];
    if (src < 0 || (uint32_t)src >= n_comp) { out[i] = 0.0f; return; }
    const uint32_t row = (uint32_t)src;
    const uint32_t n_nope = head_dim - n_rot;
    if (d < n_nope) {
        const uint8_t b = nope[(uint64_t)row * n_nope + d];
        const int k = (int)expo[(uint64_t)row * 8u + (d >> 6)];
        out[i] = dsv4_e4m3fn_decode_byte_dev(b) * exp2f((float)k);
    } else {
        out[i] = __half2float(rope[(uint64_t)row * n_rot + (d - n_nope)]);
    }
}
```

- [ ] **Step 1: Add the encode helper + commit kernel + extern (ds4_cuda.cu, after `ds4_gpu_tensor_copy_f32_to_f16` ~3750).** First the encode helper (mirrors `dsv4_e4m3fn_dequant_dev`'s binary search + tie-break exactly, returning the byte instead of the dequantized float):

```c
/* Encode an already-scaled value (|x| clamped to 448) to the model's E4M3FN byte:
 * bit7 = sign, bits0..6 = the magnitude index `best` dsv4_e4m3fn_dequant_dev would pick. */
__device__ static uint8_t dsv4_e4m3fn_encode_dev(float x) {
    uint8_t sign = (x < 0.0f) ? 0x80u : 0x00u;
    float ax = fminf(fabsf(x), 448.0f);
    int lo = 0, hi = 126;
    while (lo < hi) {
        int mid = (lo + hi + 1) >> 1;
        if (dsv4_e4m3fn_value_dev(mid) <= ax) lo = mid; else hi = mid - 1;
    }
    int best = lo;
    if (best < 126) {
        float bd = fabsf(ax - dsv4_e4m3fn_value_dev(best));
        float nd = fabsf(ax - dsv4_e4m3fn_value_dev(best + 1));
        if (nd < bd || (nd == bd && (((best + 1) & 1) == 0) && ((best & 1) != 0))) best++;
    }
    return sign | (uint8_t)best;
}
```

Then the commit kernel + extern:

```c
/* Commit `rows` staged F32 comp rows (head_dim each) to the FP8-split cache.  Re-runs the
 * model's per-64-group quantizer (fp8_kv_quantize_kernel math) emitting (E4M3 byte, int8
 * exponent) for the 448 NoPE dims and F16 for the 64 RoPE tail.  Value-exact: decode
 * reproduces the staged (already-QAT'd) value bit-for-bit.  One block/row, 64 threads. */
__global__ static void quantize_comp_f32_to_fp8split_kernel(
        uint8_t *nope_out, int8_t *exp_out, __half *rope_out, uint64_t first_row,
        const float *src, uint32_t rows, uint32_t head_dim, uint32_t n_rot) {
    const uint32_t row = blockIdx.x;
    const uint32_t tid = threadIdx.x;                 /* 0..63 */
    if (row >= rows) return;
    const uint32_t n_nope = head_dim - n_rot;         /* 448 */
    const float *xr = src + (uint64_t)row * head_dim;
    uint8_t *nr = nope_out + ((uint64_t)first_row + row) * n_nope;
    int8_t  *er = exp_out  + ((uint64_t)first_row + row) * 8u;
    __half  *rr = rope_out + ((uint64_t)first_row + row) * n_rot;
    __shared__ float scratch[64];
    uint32_t gi = 0;
    for (uint32_t off = 0; off < n_nope; off += 64u, gi++) {
        float v = (off + tid < n_nope) ? xr[off + tid] : 0.0f;
        scratch[tid] = (off + tid < n_nope) ? fabsf(v) : 0.0f;
        __syncthreads();
        for (uint32_t stride = 32; stride > 0; stride >>= 1) {
            if (tid < stride) scratch[tid] = fmaxf(scratch[tid], scratch[tid + stride]);
            __syncthreads();
        }
        int k = (int)ceilf(log2f(fmaxf(scratch[0], 1.0e-4f) / 448.0f));
        float scale = exp2f((float)k);
        if (off + tid < n_nope)
            nr[off + tid] = dsv4_e4m3fn_encode_dev(fminf(448.0f, fmaxf(-448.0f, v / scale)));
        if (tid == 0) er[gi] = (int8_t)k;
        __syncthreads();
    }
    for (uint32_t d = tid; d < n_rot; d += 64u)
        rr[d] = __float2half(xr[n_nope + d]);
}

extern "C" int ds4_gpu_tensor_quantize_f32_to_fp8split(
        ds4_gpu_tensor *nope, ds4_gpu_tensor *expo, ds4_gpu_tensor *rope, uint64_t first_row,
        const ds4_gpu_tensor *src_stage, uint32_t rows, uint32_t head_dim, uint32_t n_rot) {
    if (!nope || !expo || !rope || !src_stage) return 0;
```
